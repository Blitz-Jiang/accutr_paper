\footnotesize
\begin{frame}[allowframebreaks]{Existing Literature}
Almost all existing researches focus on designing TR variants to find an $\epsilon$-approximate stationary points for nonconvex optimization with global complexity of $O(\epsilon^{-3/2})$:
\begin{itemize}
    \item Fixed radius TR:~\citet{luenberger_linear_2021}
    \item TRACE:~\citet{curtis2017trust}
    \item Consistently adaptive trust-region:~\citet{hamad2022consistently,hamad2024simplepracticaladaptivetrustregion}
    \item Trust-Region NewtonCG:~\citet{curtis2021trust}
\end{itemize}
\framebreak
For convex optimization, \citet{jiang2024nonconvexityuniversaltrustregionmethod} proposed the universal trust-region(UTR) method, which has a complexity of $O(\epsilon^{-1/2})$ for convex optimization, the UTR has the following feature:
\begin{itemize}
    \item Its subproblem uses both trust-region ball and gradient regularization:
    \begin{equation}\label{eq.newTR}
    \begin{split}
        \min_d ~&  \frac{1}{2}d^T\left (\nabla^2 f(x) + \sigma \|\nabla f(x)\|^{1/2} I \right ) d + \nabla f(x)^T d\\
        \text{s.t. } ~& \|d\| \le r\|\nabla f(x)\|^{1/2}, \qquad \sigma, r > 0.
    \end{split}
\end{equation}
    \item The most important descent property is that each step of UTR either decreases the function value or the gradient norm sufficiently:
    \begin{equation}
        \label{eq.FOSD}
        f(x+d) - f(x) \leq -\Omega(\|\nabla f(x)\|^{1/2}), \ \text{or } \|\nabla f(x+d)\|\leq \xi \|\nabla f(x)\|, \ \xi<1.
    \end{equation}
\end{itemize}
A further question is: 
\begin{center}
\textit{can UTR be further accelerated for convex problems?}
\end{center}
\end{frame}
\begin{frame}[allowframebreaks]{Acceleration Techniques}
    Nesterov showed that the estimating sequence technique can be used to accelerate cubic Newton method~\citet{nesterov_lectures_2018}.
    \begin{itemize}
        \item The key of the technique is to maintain an estimating sequence $\phi_k(x)$ of the function with 
        \begin{equation}
            \label{eq.estimating sequence}
            A_k f(x_k) \leq \phi_k^*:=\min_x \phi_k(x) \leq A_k f(x) + \phi_0(x)
        \end{equation}
        \item A direct consequence of such sequence is 
        \begin{equation}
            \label{eq.approximate xk}
            f(x_k)-f^* \leq \frac{1}{A_k} \phi_0(x^*).
        \end{equation}
        \item Therefore it remains to analyze the growth of $A_k$.
        \framebreak
        \item Currently, cubic Newton method is the only second order method can be accelerated by the estimating sequence.
        \item There are other kind of estimating sequence(\citet{monteiro2013accelerated}), which can be used to accelerate some second-order oracles(\citet{carmon2022optimal}). However, trust-region method is not included.
    \end{itemize}
\end{frame}
\begin{frame}[allowframebreaks]{Our Method}
    In order to accelerate the trust-region method, we consider a more general UTR subproblem:
    \begin{itemize}
        \item \begin{equation}\label{eq.newTR general}
    \begin{split}
        \min_d ~&  \frac{1}{2}d^T\left (\nabla^2 f(x) + \sigma I \right ) d + \nabla f(x)^T d\\
        \text{s.t. } ~& \|d\| \le r, \qquad \sigma, r > 0.
    \end{split}
    \end{equation}
    \item We allow a more flexible choice of $\sigma$ and $r$.
    \end{itemize}
    \framebreak
    In the scenario of acceleration, the analysis for UTR no longer holds,
    \begin{itemize}
        \item In the analysis for UTR, we use the FOSD property, when the case
        \begin{equation*}
            \|\nabla f(x_k+d_k)\| \leq \xi \|\nabla f(x_k)\|
        \end{equation*}
        occurs for a logarithmic times, we are guaranteed to find approximate solutions.
        \item The reason is the relation recursively holds
        \begin{equation*}
            \|\nabla f(x_{k+j})\| \leq \xi \|\nabla f(x_{k+j-1})\| \leq \cdots\leq \xi^j \|\nabla f(x_k)\|.
        \end{equation*}
        \item However, in the acceleration scenario, we need to call UTR oracle at 
        \begin{equation*}
            x_{k+1} = \operatorname{UTR}(y_k,\sigma_k,r_k)，
        \end{equation*}
        thus the recurrence breaks down.
        \item Also, the constrained nature of the TR subproblem makes the acceleration difficult.
    \end{itemize}
    \framebreak
    \begin{itemize}
        \item The above difficulty can be addressed by properly setting $\sigma,r$.
        \item Our accelerated trust-region method shows a complexity of $\tilde O(\epsilon^{-1/3})$ for convex optimization.
    \end{itemize}
\end{frame}
